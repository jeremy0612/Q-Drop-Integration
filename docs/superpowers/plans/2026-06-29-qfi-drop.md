# QFI-Drop Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the no-op `--algorithm pruning` with QFI-Drop — per-gate pruning scored by Quantum-Fisher spectral leverage, with a unified natural-gradient step on the surviving sub-manifold — for the Quantum GCN, plus a QAdaPrune baseline to compare against.

**Architecture:** A new `--algorithm qfi` lives entirely inside the existing Q-Drop hook framework (`start_epoch` probes the data-averaged QFIM once per epoch; `after_backward` freezes low-leverage gates and QNG-preconditions the survivors). No optimizer swap, no training-loop change. A parallel `--algorithm qadaprune` implements the gradient-difference baseline in the same hooks.

**Tech Stack:** PyTorch, PennyLane (`qml.metric_tensor`), PyTorch Geometric, pytest. Reuses `qml_techniques/qng.py` (`apply_fubini_study_precondition`).

**Spec:** `docs/superpowers/specs/2026-06-29-qfi-drop-design.md`

**Planning refinement (supersedes spec §3.2):** the spec's `keep_quantile=0.33` is replaced by an absolute `energy_threshold=0.5` — freeze a gate when the majority of its energy lies in null eigendirections. This removes one hyperparameter and avoids quantile degeneracy when many leverage scores are tied at zero (discovered while writing the Task 3 tests).

```python
# src/qdrop/qfi_metric.py
"""Data-averaged Quantum Fisher Information Matrix for a QuantumCircuitAdapter.

The QFIM is the Fubini-Study metric (x4) of the node-embedding circuit. It is
input-dependent (AngleEmbedding), so we average it over a batch of node-feature
vectors. Computed under no_grad and treated as a constant preconditioner.
"""
from __future__ import annotations

import pennylane as qml
import torch


def _metric_for_one_input(qnode, x: torch.Tensor, weights: torch.Tensor) -> torch.Tensor:
    # metric_tensor differentiates w.r.t. trainable args. `weights` is the only
    # trainable arg; `x` (data) must be non-trainable so the transform targets
    # weights. approx="block-diag" is the cheap, well-conditioned default.
    mt = qml.metric_tensor(qnode, approx="block-diag")(x, weights)
    return mt.reshape(weights.numel(), weights.numel())


def compute_qfim(adapter, probe_inputs: torch.Tensor) -> torch.Tensor:
    """probe_inputs: (B, n_qubits) bounded angle vectors. Returns (p, p) mean QFIM."""
    qnode = adapter.quantum_layer.qnode
    weights = adapter.weights.detach()
    mats = []
    with torch.no_grad():
        for i in range(probe_inputs.shape[0]):
            x = probe_inputs[i].detach()
            mats.append(_metric_for_one_input(qnode, x, weights))
    # Fubini-Study -> QFIM is a constant factor (4); irrelevant to leverage ranking, omit.
    return torch.stack(mats).mean(dim=0)
```

```python
# src/qdrop/qfi.py
"""Framework-agnostic QFI-Drop math: spectral leverage and the unified update."""
from __future__ import annotations

import torch


def spectral_leverage(F: torch.Tensor, spectral_ratio: float = 1e-3) -> torch.Tensor:
    """Per-gate kept-energy e_j = sum_{i: lambda_i >= ratio*lambda_max} U_ji^2, in [0, 1].

    F: (p, p) symmetric PSD QFIM. Returns (p,) leverage scores.
    """
    F = 0.5 * (F + F.T)                                  # symmetrize numerical noise
    evals, evecs = torch.linalg.eigh(F)                  # ascending; evecs columns
    lam_max = evals.max().clamp_min(torch.finfo(F.dtype).tiny)
    keep = evals >= spectral_ratio * lam_max             # (p,) over eigen-index
    return (evecs[:, keep] ** 2).sum(dim=1)              # row sum over kept columns
```

```python
# append to src/qdrop/qfi.py
from qml_techniques.qng import apply_fubini_study_precondition


def prune_and_precondition(
    grad: torch.Tensor,
    F: torch.Tensor,
    e: torch.Tensor,
    energy_threshold: float = 0.5,
    reg: float = 1e-4,
) -> torch.Tensor:
    """Freeze gates with leverage < energy_threshold; QNG-precondition survivors.

    grad: weight gradient (any shape, p elements). F: (p, p) QFIM. e: (p,) leverage.
    Returns the modified gradient, same shape as grad.
    """
    g = grad.reshape(-1)
    survive = e >= energy_threshold
    out = torch.zeros_like(g)
    if survive.any():
        idx = survive.nonzero(as_tuple=True)[0]
        F_ss = F.index_select(0, idx).index_select(1, idx)
        out[idx] = apply_fubini_study_precondition(g[idx], F_ss, reg=reg)
    return out.reshape(grad.shape)
```

```python
# src/qdrop/qadaprune.py
"""QAdaPrune baseline (arXiv:2408.13352): gradient-difference importance + freeze."""
from __future__ import annotations

import torch


class QAdaPruneState:
    def __init__(self, n: int, window: int = 10, keep: float = 0.75):
        self.window = window
        self.keep = keep
        self.score = torch.zeros(n)
        self._prev = None
        self._count = 0

    def observe(self, grad_flat: torch.Tensor) -> None:
        g = grad_flat.reshape(-1).detach()
        if self._prev is not None:
            self.score += (g - self._prev).abs()
        self._prev = g.clone()
        self._count += 1

    def keep_mask(self) -> torch.Tensor:
        n = self.score.numel()
        k = max(1, int(self.keep * n))
        top = torch.topk(self.score, k).indices
        mask = torch.zeros(n, dtype=torch.bool)
        mask[top] = True
        return mask

    def apply(self, grad: torch.Tensor) -> torch.Tensor:
        mask = self.keep_mask().reshape(grad.shape)
        return torch.where(mask, grad, torch.zeros_like(grad))
```
